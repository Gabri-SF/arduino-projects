#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <PubSubClient.h>
#include <math.h>

// ==========================================
// 1. CONFIGURAÇÕES (ALTERA AQUI)
// ==========================================
const char* ssid = "";
const char* password = "";

// Credenciais HiveMQ (Devem ser as mesmas do teu .env.local)
const char* mqtt_server = "";
const int mqtt_port = 8883;
const char* mqtt_user = "";
const char* mqtt_pass = "";

// Tópicos
const char* topic_luz_sala = "home/sala/luz";
const char* topic_luz_cozinha = "home/cozinha/luz";
const char* topic_led_quarto = "home/quarto/led";
const char* topic_temp = "home/sensor/temperatura";
const char* topic_status = "home/status/esp32";

// ==========================================
// 2. DEFINIÇÕES DE PINS
// ==========================================
const int PIN_SALA = D8;      // Switch
const int PIN_COZINHA = D9;   // Switch
const int PIN_QUARTO = D10;   // Slider (PWM)
const int THERMISTOR_PIN = A0;

// Configurações PWM para o LED D10
const int PWM_CHANNEL = 0;
const int PWM_FREQ = 5000;
const int PWM_RES = 8;

// Constantes Termistor
const float SERIES_RESISTOR = 10000.0;
const float NOMINAL_RESISTANCE = 10000.0;
const float NOMINAL_TEMPERATURE = 25.0;
const float B_COEFFICIENT = 3380.0;

// Filtro EMA
float filteredTemp = 25.0;
const float ALPHA = 0.1;
unsigned long lastTempMsg = 0;

WiFiClientSecure espClient;
PubSubClient client(espClient);

// ==========================================
// 3. LÓGICA MQTT (RECEBER COMANDOS)
// ==========================================
void callback(char* topic, byte* payload, unsigned int length) {
  String message;
  for (int i = 0; i < length; i++) message += (char)payload[i];

  Serial.print("Mensagem recebida [");
  Serial.print(topic);
  Serial.print("]: ");
  Serial.println(message);

  if (String(topic) == topic_luz_sala) {
    digitalWrite(PIN_SALA, message == "ON" ? HIGH : LOW);
  }
  else if (String(topic) == topic_luz_cozinha) {
    digitalWrite(PIN_COZINHA, message == "ON" ? HIGH : LOW);
  }
  else if (String(topic) == topic_led_quarto) {
    int val = message.toInt(); // 0 a 100
    int brightness = map(val, 0, 100, 0, 255);
    ledcWrite(PWM_CHANNEL, brightness);
  }
}

// ==========================================
// 4. LÓGICA DE TEMPERATURA
// ==========================================
float readTemperature() {
  int adcValue = analogRead(THERMISTOR_PIN);
  float voltage = adcValue / 4095.0 * 3.3;
  if (voltage <= 0.01) return -100.0;

  float resistance = SERIES_RESISTOR * (3.3 / voltage - 1.0);
  float temperature = resistance / NOMINAL_RESISTANCE;
  temperature = log(temperature);
  temperature /= B_COEFFICIENT;
  temperature += 1.0 / (NOMINAL_TEMPERATURE + 273.15);
  temperature = 1.0 / temperature;

  return temperature - 273.15;
}

void reconnect() {
  while (!client.connected()) {
    Serial.print("A tentar ligação MQTT...");
    if (client.connect("ESP32_Nano_Client", mqtt_user, mqtt_pass, topic_status, 1, true, "OFFLINE")) {
      Serial.println("Ligado!");
      client.publish(topic_status, "ONLINE", true);
      client.subscribe(topic_luz_sala);
      client.subscribe(topic_luz_cozinha);
      client.subscribe(topic_led_quarto);
    } else {
      Serial.print("Falhou, rc=");
      Serial.print(client.state());
      Serial.println(" a tentar novamente em 5s");
      delay(5000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);

  pinMode(PIN_SALA, OUTPUT);
  pinMode(PIN_COZINHA, OUTPUT);

  // Setup PWM para o D10
  ledcSetup(PWM_CHANNEL, PWM_FREQ, PWM_RES);
  ledcAttachPin(PIN_QUARTO, PWM_CHANNEL);

  // WiFi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); }
  Serial.println("\nWiFi Ligado!");

  // Configurar SSL/TLS para HiveMQ
  espClient.setInsecure();

  client.setServer(mqtt_server, mqtt_port);
  client.setCallback(callback);
}

void loop() {
  if (!client.connected()) reconnect();
  client.loop();

  // Enviar temperatura a cada 2 segundos
  unsigned long now = millis();
  if (now - lastTempMsg > 2000) {
    lastTempMsg = now;

    float currentRawTemp = readTemperature();
    filteredTemp = (ALPHA * currentRawTemp) + (1.0 - ALPHA) * filteredTemp;

    char tempString[8];
    dtostrf(filteredTemp, 1, 2, tempString);
    client.publish(topic_temp, tempString);

    Serial.print("Temp enviada: ");
    Serial.println(tempString);
  }
}
